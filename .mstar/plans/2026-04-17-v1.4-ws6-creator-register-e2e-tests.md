## Task 2 — Happy-path E2E test

- [ ] **Step 1: Write test**

In `crates/nexus42/tests/creator_register_e2e.rs`:

```rust
use wiremock::{matchers::{method, path}, Mock, MockServer, ResponseTemplate};

#[tokio::test]
async fn creator_register_happy_path() {
    let mock = MockServer::start().await;

    // Mock POST /creators/register → returns a solvable challenge.
    Mock::given(method("POST")).and(path("/creators/register"))
        .respond_with(ResponseTemplate::new(200).set_body_string(include_str!("fixtures/creator_register/register_response.json")))
        .mount(&mock).await;

    // Mock POST /creators/verify → returns api_key.
    Mock::given(method("POST")).and(path("/creators/verify"))
        .respond_with(ResponseTemplate::new(200).set_body_string(include_str!("fixtures/creator_register/verify_response_success.json")))
        .mount(&mock).await;

    // Point the CLI at the mock:
    let bin = env!("CARGO_BIN_EXE_nexus42");
    let home = tempfile::tempdir().unwrap();
    let output = std::process::Command::new(bin)
        .args(["creator", "register", "--name", "test-creator"])
        .env("HOME", home.path())
        .env("NEXUS42_PLATFORM_URL", &mock.uri())
        .output().expect("run nexus42");

    assert!(output.status.success(),
        "register exited non-zero:\nSTDOUT: {}\nSTDERR: {}",
        String::from_utf8_lossy(&output.stdout),
        String::from_utf8_lossy(&output.stderr));

    // Assert the api_key was stored. Location depends on how VR3's storage settled —
    // either `$HOME/.nexus42/auth.json` or an equivalent creds file:
    let creds = std::fs::read_to_string(home.path().join(".nexus42/auth.json"))
        .expect("auth.json should exist after register");
    assert!(creds.contains("creator_api_key"), "creds file missing api_key: {creds}");
}
```

Run: `cargo test -p nexus42 --test creator_register_e2e creator_register_happy_path -- --nocapture`
Expected: PASS (assuming challenge-solver + PlatformClient + storage are all already implemented, which V1.3 confirmed).

If it fails due to the challenge fixture being unsolvable with the real solver, either regenerate the fixture to match the solver's grammar or stub the solver via an env var — document the choice in the fixture file's header comment.
