use std::path::PathBuf;

/// Local API transport configuration
#[derive(Debug, Clone)]
pub enum Transport {
    /// HTTP over TCP loopback (default)
    Http { port: u16 },
    /// Unix domain socket
    UnixSocket { path: PathBuf },
}

impl Transport {
    /// Load from environment
    pub fn from_env() -> Self {
        if let Ok(path) = std::env::var("NEXUS_DAEMON_SOCKET_PATH") {
            Transport::UnixSocket { path: PathBuf::from(path) }
        } else {
            let port = std::env::var("NEXUS_DAEMON_PORT")
                .ok()
                .and_then(|s| s.parse().ok())
                .unwrap_or(8420);
            Transport::Http { port }
        }
    }
}
```

pub async fn start_server(transport: Transport) -> Result<(),ServerError> {
    let app = create_router(pool);
    
    match transport {
        Transport::Http { port } => {
            let addr = SocketAddr::from(([127, 0, 0, 1], port));
            tracing::info!(%addr, "Starting HTTP server");
            
            let listener = TcpListener::bind(addr).await?;
            axum::serve(listener, app).await?;
        }
        Transport::UnixSocket { path } => {
            tracing::info!(?path, "Starting Unix socket server");
            
            // Remove existing socket if present
            if path.exists() {
                std::fs::remove_file(&path)?;
            }
            
            let listener = UnixListener::bind(&path)?;
            axum::serve(listener, app).await?;
        }
    }
    
    Ok(())
}

## Local API Transport

The daemon supports two transport mechanisms for the Local API:

### HTTP (Default)

```bash
# Default: HTTP on localhost:8420
nexus42 daemon start

# Custom port
NEXUS_DAEMON_PORT=9000 nexus42 daemon start
```

### Unix Socket (Recommended for production)

```bash
# Use Unix socket for better security and performance
NEXUS_DAEMON_SOCKET_PATH=/tmp/nexus42d.sock nexus42 daemon start
```

**Benefits of Unix socket:**
- Better security (no network exposure)
- Lower latency (no TCP overhead)
- Filesystem permissions control

**When to use HTTP:**
- Development/debugging (easier to test with curl)
- Cross-machine communication (future)
- Windows compatibility (Unix sockets not available)

use serde::{Deserialize, Serialize};
use chrono::{DateTime, Utc};

/// Persistent session data
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Session {
    pub id: String,
    pub agent_id: String,
    pub created_at: DateTime<Utc>,
    pub last_active: DateTime<Utc>,
    pub expires_at: DateTime<Utc>,
    pub metadata: serde_json::Value,
}

/// Session store backed by SQLite
pub struct SessionStore {
    pool: DbPool,
}

impl SessionStore {
    pub fn new(pool: DbPool) -> Self {
        Self { pool }
    }
    
    /// Initialize sessions table
    pub async fn initialize(&self) -> Result<(), SessionError> {
        self.pool.get().await?.execute(
            "CREATE TABLE IF NOT EXISTS sessions (
                id TEXT PRIMARY KEY,
                agent_id TEXT NOT NULL,
                created_at TEXT NOT NULL,
                last_active TEXT NOT NULL,
                expires_at TEXT NOT NULL,
                metadata TEXT NOT NULL DEFAULT '{}'
            )",
            [],
        ).await?;
        
        Ok(())
    }
    
    /// Create a new session
    pub async fn create(
        &self,
        agent_id: &str,
        ttl: Duration,
    ) -> Result<Session, SessionError> {
        let session = Session {
            id: generate_session_id(),
            agent_id: agent_id.to_string(),
            created_at: Utc::now(),
            last_active: Utc::now(),
            expires_at: Utc::now() + chrono::Duration::from_std(ttl)?,
            metadata: serde_json::json!({}),
        };
        
        self.pool.get().await?.execute(
            "INSERT INTO sessions (id, agent_id, created_at, last_active, expires_at, metadata)
             VALUES (?, ?, ?, ?, ?, ?)",
            [
                &session.id,
                &session.agent_id,
                &session.created_at.to_rfc3339(),
                &session.last_active.to_rfc3339(),
                &session.expires_at.to_rfc3339(),
                &session.metadata.to_string(),
            ],
        ).await?;
        
        Ok(session)
    }
    
    /// Get session by ID
    pub async fn get(&self, id: &str) -> Result<Option<Session>, SessionError> {
        let row = self.pool.get().await?.query_row(
            "SELECT id, agent_id, created_at, last_active, expires_at, metadata
             FROM sessions WHERE id = ?",
            [id],
            |row| {
                Ok(Session {
                    id: row.get(0)?,
                    agent_id: row.get(1)?,
                    created_at: DateTime::parse_from_rfc3339(&row.get::<_, String>(2)?)
                        .unwrap()
                        .with_timezone(&Utc),
                    last_active: DateTime::parse_from_rfc3339(&row.get::<_, String>(3)?)
                        .unwrap()
                        .with_timezone(&Utc),
                    expires_at: DateTime::parse_from_rfc3339(&row.get::<_, String>(4)?)
                        .unwrap()
                        .with_timezone(&Utc),
                    metadata: serde_json::from_str(&row.get::<_, String>(5)?).unwrap(),
                })
            },
        ).optional()?;
        
        // Check expiration
        if let Some(ref session) = row {
            if session.expires_at < Utc::now() {
                self.delete(id).await?;
                return Ok(None);
            }
        }
        
        Ok(row)
    }
    
    /// Update session last_active timestamp
    pub async fn touch(&self, id: &str) -> Result<(), SessionError> {
        self.pool.get().await?.execute(
            "UPDATE sessions SET last_active = ? WHERE id = ?",
            [Utc::now().to_rfc3339(), id.to_string()],
        ).await?;
        
        Ok(())
    }
    
    /// Delete session
    pub async fn delete(&self, id: &str) -> Result<(), SessionError> {
        self.pool.get().await?.execute(
            "DELETE FROM sessions WHERE id = ?",
            [id],
        ).await?;
        
        Ok(())
    }
    
    /// Clean up expired sessions
    pub async fn cleanup_expired(&self) -> Result<u64, SessionError> {
        let count = self.pool.get().await?.execute(
            "DELETE FROM sessions WHERE expires_at < ?",
            [Utc::now().to_rfc3339()],
        ).await?;
        
        Ok(count)
    }
}

fn generate_session_id() -> String {
    use rand::Rng;
    const CHARSET: &[u8] = b"abcdefghijklmnopqrstuvwxyz0123456789";
    let mut rng = rand::thread_rng();
    
    (0..32)
        .map(|_| {
            let idx = rng.gen_range(0..CHARSET.len());
            CHARSET[idx] as char
        })
        .collect()
}

use serde::{Deserialize, Serialize};
use std::collections::HashSet;

/// Permission types for ACP operations
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
pub enum Permission {
    /// Read world data
    WorldRead,
    /// Write world data (create, update)
    WorldWrite,
    /// Delete world data
    WorldDelete,
    /// Execute sync operations
    SyncExecute,
    /// Access context assembly
    ContextAccess,
    /// Manage sessions
    SessionManage,
    /// Execute agent skills
    SkillExecute,
    /// Admin operations
    Admin,
}

/// Permission set for an agent or session
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct PermissionSet {
    pub granted: HashSet<Permission>,
    pub denied: HashSet<Permission>,
}

impl PermissionSet {
    /// Create default permissions for a new agent
    pub fn default_agent() -> Self {
        let mut granted = HashSet::new();
        granted.insert(Permission::WorldRead);
        granted.insert(Permission::ContextAccess);
        granted.insert(Permission::SkillExecute);
        
        Self {
            granted,
            denied: HashSet::new(),
        }
    }
    
    /// Create admin permissions
    pub fn admin() -> Self {
        let mut granted = HashSet::new();
        granted.insert(Permission::Admin);
        
        Self {
            granted,
            denied: HashSet::new(),
        }
    }
    
    /// Check if a permission is granted
    pub fn has(&self, permission: &Permission) -> bool {
        // Denied takes precedence
        if self.denied.contains(permission) {
            return false;
        }
        
        // Admin has all permissions
        if self.granted.contains(&Permission::Admin) {
            return true;
        }
        
        self.granted.contains(permission)
    }
    
    /// Grant a permission
    pub fn grant(&mut self, permission: Permission) {
        self.denied.remove(&permission);
        self.granted.insert(permission);
    }
    
    /// Deny a permission
    pub fn deny(&mut self, permission: Permission) {
        self.granted.remove(&permission);
        self.denied.insert(permission);
    }
}
```

use super::{Permission, PermissionSet};

/// Permission checker for ACP operations
pub struct PermissionChecker {
    /// Session permissions cache
    session_permissions: std::collections::HashMap<String, PermissionSet>,
}

impl PermissionChecker {
    pub fn new() -> Self {
        Self {
            session_permissions: std::collections::HashMap::new(),
        }
    }
    
    /// Check if session has permission
    pub fn check(
        &self,
        session_id: &str,
        required: &Permission,
    ) -> Result<(), PermissionError> {
        let permissions = self.session_permissions
            .get(session_id)
            .ok_or(PermissionError::SessionNotFound(session_id.to_string()))?;
        
        if !permissions.has(required) {
            return Err(PermissionError::InsufficientPermissions {
                session_id: session_id.to_string(),
                required: required.clone(),
            });
        }
        
        Ok(())
    }
    
    /// Set permissions for a session
    pub fn set_permissions(
        &mut self,
        session_id: &str,
        permissions: PermissionSet,
    ) {
        self.session_permissions.insert(session_id.to_string(), permissions);
    }
    
    /// Remove permissions for a session
    pub fn remove_permissions(&mut self, session_id: &str) {
        self.session_permissions.remove(session_id);
    }
}

#[derive(Debug, thiserror::Error)]
pub enum PermissionError {
    #[error("Session not found: {0}")]
    SessionNotFound(String),
    
    #[error("Insufficient permissions: {required:?} required for session {session_id}")]
    InsufficientPermissions {
        session_id: String,
        required: Permission,
    },
}

use colored::Colorize;
use std::fmt;

/// User-friendly CLI error
pub struct CliError {
    /// Error code for documentation
    pub code: &'static str,
    /// User-friendly message
    pub message: String,
    /// Suggested fix
    pub suggestion: Option<String>,
    /// Underlying error (for debugging)
    pub source: Option<anyhow::Error>,
}

impl CliError {
    /// Create a new CLI error
    pub fn new(code: &'static str, message: impl Into<String>) -> Self {
        Self {
            code,
            message: message.into(),
            suggestion: None,
            source: None,
        }
    }
    
    /// Add a suggestion
    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
        self.suggestion = Some(suggestion.into());
        self
    }
    
    /// Add source error
    pub fn with_source(mut self, source: impl Into<anyhow::Error>) -> Self {
        self.source = Some(source.into());
        self
    }
}

impl fmt::Display for CliError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{} {}", "Error".red().bold(), self.code)?;
        write!(f, "\n\n{}", self.message)?;
        
        if let Some(suggestion) = &self.suggestion {
            write!(f, "\n\n{} {}", "Suggestion:".yellow().bold(), suggestion)?;
        }
        
        Ok(())
    }
}

impl fmt::Debug for CliError {
    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
        write!(f, "{}", self)?;
        if let Some(source) = &self.source {
            write!(f, "\n\nCaused by:\n{}", source)?;
        }
        Ok(())
    }
}
