- [ ] **Step 4: Reuse a static Handlebars registry for template rendering**

Use `std::sync::OnceLock` or the repository's existing equivalent. Keep helper behavior unchanged:

```rust
use std::sync::OnceLock;

static HANDLEBARS: OnceLock<handlebars::Handlebars<'static>> = OnceLock::new();

fn handlebars_registry() -> &'static handlebars::Handlebars<'static> {
    HANDLEBARS.get_or_init(handlebars::Handlebars::new)
}

fn render_core_context_template(
    template: &str,
    payload: &serde_json::Value,
) -> anyhow::Result<String> {
    handlebars_registry()
        .render_template(template, payload)
        .map_err(Into::into)
}
```
