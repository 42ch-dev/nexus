```rust
use std::path::{Path, PathBuf};

/// Validate that a path stays within the allowed base directory
/// 
/// Prevents path traversal attacks like "../../../etc/passwd"
pub fn validate_path_within_base(path: &Path, base: &Path) -> Result<PathBuf, SummaryError> {
    // Canonicalize both paths to resolve symlinks and .. 
    let canonical_path = path.canonicalize().map_err(|_| {
        SummaryError::InvalidPath(format!("Cannot canonicalize: {}", path.display()))
    })?;
    
    let canonical_base = base.canonicalize().map_err(|_| {
        SummaryError::InvalidPath(format!("Cannot canonicalize base: {}", base.display()))
    })?;
    
    // Check if path starts with base
    if !canonical_path.starts_with(&canonical_base) {
        return Err(SummaryError::PathTraversal {
            path: path.display().to_string(),
            base: base.display().to_string(),
        });
    }
    
    Ok(canonical_path)
}

/// Error types for summary generation
#[derive(Debug, thiserror::Error)]
pub enum SummaryError {
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
    
    #[error("Invalid path: {0}")]
    InvalidPath(String),
    
    #[error("Path traversal detected: {path} is outside {base}")]
    PathTraversal { path: String, base: String },
    
    #[error("UTF-8 error: {0}")]
    Utf8(#[from] std::str::Utf8Error),
}
```

```rust
/// Recursively scan directory for files
pub fn scan_recursive(
    base: &Path,
    current: &Path,
    config: &SummaryConfig,
    depth: usize,
) -> Result<Vec<FileSummary>, SummaryError> {
    // Validate current path is within base
    let validated_current = validate_path_within_base(current, base)?;
    
    // Check depth limit
    if depth > config.max_depth {
        tracing::debug!(path = %validated_current.display(), "Max depth reached");
        return Ok(vec![]);
    }
    
    let mut summaries = Vec::new();
    
    for entry in std::fs::read_dir(&validated_current)? {
        let entry = entry?;
        let path = entry.path();
        
        // Validate each entry is within base
        let validated_path = validate_path_within_base(&path, base)?;
        
        if validated_path.is_dir() {
            // Recurse into subdirectory
            let sub_summaries = scan_recursive(base, &validated_path, config, depth + 1)?;
            summaries.extend(sub_summaries);
        } else if validated_path.is_file() {
            // Generate summary for file
            if let Some(summary) = generate_summary(&validated_path, config)? {
                summaries.push(summary);
            }
        }
    }
    
    Ok(summaries)
}
```

```rust
mod truncation_tests {

    fn test_truncate_ascii() {
        let text = "Hello World";
        let result = truncate_at_char_boundary(text, 5);
        assert_eq!(result, "Hello");
        assert!(result.is_char_boundary(result.len()));

    #[test]
    fn test_truncate_chinese() {
        // Chinese characters are 3 bytes each
        let text = "你好世界"; // 12 bytes
        let result = truncate_at_char_boundary(text, 7);

    }

        let text = "👋🌍🚀"; // 12+ bytes
        let result = truncate_at_char_boundary(text, 5);
        // Should not split emoji

    #[test]
    fn test_truncate_mixed() {
        let text = "Hello 世界 🚀 End";
        let result = truncate_at_char_boundary(text, 12);
        assert!(result.is_char_boundary(result.len()));

        assert_eq!(result, "");
    }
    
    #[test]
    fn test_truncate_exact_boundary() {
        let text = "Test";
        let result = truncate_at_char_boundary(text, 4);
        assert_eq!(result, "Test");
    }
    
    #[test]
    fn test_truncate_beyond_length() {
        let text = "Short";
        let result = truncate_at_char_boundary(text, 100);
        assert_eq!(result, "Short");

/// Truncate string at a safe character boundary
fn truncate_at_char_boundary(s: &str, max_bytes: usize) -> &str {
    if s.len() <= max_bytes {
        return s;
    }
    
    // Find the last valid char boundary before max_bytes
    let mut boundary = max_bytes;
    while boundary > 0 && !s.is_char_boundary(boundary) {
        boundary -= 1;
    }
    
    &s[..boundary]
}
```

```rust
/// ## SQLite File Locking Behavior
/// 
/// ### WAL Mode (Recommended)
/// 
/// SQLite in WAL (Write-Ahead Logging) mode allows:
/// - Concurrent reads while writing
/// - Multiple writers (serialized internally)
/// - Better performance for read-heavy workloads
/// 
/// ### Known Limitations
/// 
/// 1. **Single writer at a time**: Even in WAL mode, only one write 
///    transaction can be active. Concurrent writes are serialized.
/// 
/// 2. **File system locks**: On some file systems (NFS, SMB), SQLite 
///    file locking may not work correctly. Always use local storage.
/// 
/// 3. **Process boundaries**: Multiple daemon processes cannot share 
///    the same database file safely.
///

///

/// - Multiple readers can work simultaneously
/// - WAL mode allows concurrent read/write
/// - Pool correctly handles connection contention
///

/// - Cross-process file locking (requires multiple daemon instances)
```

/// There's a small race condition window between `init_workspace()` 
/// completing and the middleware checking workspace state:
/// 
/// 1. Request A arrives, triggers `init_workspace()`
/// 2. `init_workspace()` completes, writes to disk
/// 3. Request B arrives before middleware cache updates
/// 4. Request B sees stale state, may re-trigger init
/// 
/// ### Why It's Acceptable
/// 
/// 1. **In-memory mutex**: The workspace state is protected by a 
///    `tokio::sync::RwLock`, so concurrent access is serialized.
/// 
/// 2. **Idempotent init**: `init_workspace()` is idempotent - 
///    calling it multiple times is safe and just overwrites with 
///    the same data.
/// 
/// 3. **Window is negligible**: The race window is microseconds 
///    (mutex release → next check), and init takes milliseconds.
/// 
/// 4. **No data loss**: Even if init runs twice, no data is lost 
///    or corrupted.
///
