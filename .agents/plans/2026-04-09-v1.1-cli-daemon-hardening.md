```rust
/// Pool configuration with sensible defaults
#[derive(Debug, Clone)]
pub struct PoolConfig {
    /// Connection timeout (default: 30s)
    pub timeout: Duration,
    /// Max connections (default: 10)
    pub max_connections: usize,
}

impl Default for PoolConfig {
    fn default() -> Self {
        Self {
            timeout: Duration::from_secs(30),
            max_connections: 10,
        }
    }
}

impl PoolConfig {
    /// Load from environment variables
    pub fn from_env() -> Self {
        let timeout = std::env::var("NEXUS_DB_POOL_TIMEOUT_SECS")
            .ok()
            .and_then(|s| s.parse().ok())
            .unwrap_or(30);
        
        let max_connections = std::env::var("NEXUS_DB_POOL_MAX_CONNECTIONS")
            .ok()
            .and_then(|s| s.parse().ok())
            .unwrap_or(10);
        
        Self {
            timeout: Duration::from_secs(timeout),
            max_connections,
        }
    }
    
    /// Builder: set timeout
    pub fn with_timeout(mut self, timeout: Duration) -> Self {
        self.timeout = timeout;
        self
    }
    
    /// Builder: set max connections
    pub fn with_max_connections(mut self, max: usize) -> Self {
        self.max_connections = max;
        self
    }
}
```

```rust
impl DeltaBundleBuilder {
    /// Validate bundle integrity before building
    /// 
    /// Checks:
    /// - Deltas are in chronological order
    /// - No duplicate delta IDs
    /// - Base versions are valid
    pub fn validate(&self) -> Result<(), BundleValidationError> {
        // Check chronological order
        let mut last_timestamp = 0u64;
        for (i, delta) in self.deltas.iter().enumerate() {
            if delta.timestamp < last_timestamp {
                return Err(BundleValidationError::NonMonotonicTimestamp {
                    index: i,
                    timestamp: delta.timestamp,
                    previous: last_timestamp,
                });
            }
            last_timestamp = delta.timestamp;
        }
        
        // Check for duplicate IDs
        let mut seen_ids = std::collections::HashSet::new();
        for delta in &self.deltas {
            if !seen_ids.insert(&delta.id) {
                return Err(BundleValidationError::DuplicateDeltaId(delta.id.clone()));
            }
        }
        
        Ok(())
    }
    
    /// Build bundle with validation (recommended)
    pub fn build_validated(self) -> Result<DeltaBundle, BundleValidationError> {
        self.validate()?;
        Ok(self.build())
    }
}
```

```rust
/// Validate auth token format before making requests
/// 
/// Checks:
/// - Token is not empty
/// - Token has minimum length (64 chars for API keys)
/// - Token contains only valid characters
fn validate_auth_token(token: &str) -> Result<(), SyncError> {
    if token.is_empty() {
        return Err(SyncError::Auth("Auth token cannot be empty".into()));
    }
    
    if token.len() < 64 {
        return Err(SyncError::Auth(
            "Auth token must be at least 64 characters".into()
        ));
    }
    
    // API keys should be alphanumeric + hyphens/underscores
    if !token.chars().all(|c| c.is_alphanumeric() || c == '-' || c == '_') {
        return Err(SyncError::Auth(
            "Auth token contains invalid characters".into()
        ));
    }
    
    Ok(())
}

impl SyncClient {
    pub fn new(base_url: &str, auth_token: &str) -> Result<Self, SyncError> {
        // Validate token format
        validate_auth_token(auth_token)?;
        
        Ok(Self {
            base_url: base_url.to_string(),
            auth_token: auth_token.to_string(),
            client: reqwest::Client::new(),
        })
    }
}
```
