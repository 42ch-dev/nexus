```rust
impl MembershipRole {
    /// Check if role can confirm canon content
    /// Per spec: owner + maintainer + official_creator can confirm
    pub fn can_confirm_canon(&self) -> bool {
        matches!(
            self,
            MembershipRole::Owner | MembershipRole::Maintainer | MembershipRole::OfficialCreator
        )
    }

    /// Check if role can sync knowledge base
    /// Per spec: owner + maintainer + collaborator can sync
    pub fn can_sync_kb(&self) -> bool {
        matches!(
            self,
            MembershipRole::Owner | MembershipRole::Maintainer | MembershipRole::Collaborator
        )
    }
}
```

```json
      "type": "string",

      "type": "string",

      "type": "object",
```
